CREATE TABLE ISSUES
(
    ID SERIAL PRIMARY KEY,
    ISSUESLIST_ID BIGINT UNSIGNED NOT NULL,
    SUMMARY VARCHAR(1000),
    DESCRIPTION VARCHAR(2000),
    USER_ID_ASSIGNEDTO BIGINT UNSIGNED NOT NULL,

        FOREIGN KEY (USER_ID_ASSIGNEDTO) REFERENCES USERS(ID)
